systemctl start nginx.service

创建虚拟主机目录

mkdir -p /var/www/frontEnd/public/
touch index.html

配置nginx

vim /etc/nginx/conf.d/vitual.conf

nginx -t 检测配置文件是否正常

linux(centos) 安装node

yum install -y wget
wget https://npm.taobao.org/mirrors/node/v8.0.0/node-v8.0.0-linux-x64.tar.xz（下载）
tar -xvf node-v8.0.0-linux-x64.tar.xz -C /opt(解压)

第一种方式

ln -s /opt/node-v8.0.0-linux-x64/bin/node /usr/bin/node (部署bin文件 环境变量)
ln -s /opt/node-v8.0.0-linux-x64/bin/npm /usr/bin/npm (部署bin文件 环境变量)

node -v
安装cnpm npm install cnpm -g –registry=https://registry.npm.taobao.org

第二种方式(推荐)

vim /etc/profile
export N_PREFIX=/opt/node-v8.0.0-linux-x64 #node实际安装位置
export PATH=$N_PREFIX/bin:$PATH
source /etc/profile
