创建dom

var p = document.creatElement(“p”);

添加文本节点

var node = document.creatTextNode(“haha”);
p.appendChild(node);

获取dom

id 获取

document.getElementsById(“div”)

通过name获取

document.getElementsByName(“name”)

通过标签名

document.getElementsByTagName(“p”);

通过类名获取

document.getElementsByClassName(“”);
ie8 以前不兼容

获取html

document.documentElement

获取body

document.body

通过选择器获取一个元素

document.querySelector(“#demo”);
document.querySelector(“.demo”);
document.querySelector(“p”);
document.querySelector(“p.example”);
document.querySelector(“a[target]”);
不兼容IE7以及以前版本

返回数组

querySelectorAll(“.demo”);
querySelectorAll(“p”);
querySelectorAll(“p.example”)
querySelectorAll(“a[target]”);
ie8 以前不兼容

####query 出的是静态的，不会随着节点增减而变化

js样式设置

直接设置style的属性 (设置 !important值无效)

element.style.height = ‘100px’;
element.style[‘text-align’] = ‘100px’;
element.style.textAlign= ‘100px’;

直接设置属性（只能用于某些属性，相关样式会自动识别）

element.setAttribute(‘height’, 100);
element.setAttribute(‘height’, ‘100px’);

设置style的属性

element.setAttribute(‘style’, ‘height: 100px !important’);

setProperty

element.style.setProperty(‘height’, ‘300px’, ‘important’);

改变class

element.className += ‘blue fb’;

设置cssText

element.style.cssText += ‘height: 100px !important’;

删除dom

必须获取父级元素

var parent = document.getElementById(“parent”);
var child = document.getElementById(“child”);
parent.removeChild(child);

不引用父元素

child.parentNode.removeChild(“child”);

事件绑定

document.getElementById(“demo”).onclick = function(){}

document.getElementById(“demo”).addEventListener(‘click’,function(){})
document.getElementById(“demo”).attachEvent(‘click’,function(){}) //ie兼容

事件委托

利用事件冒泡 解决大量绑定问题

e.srcElement //ie

js禁止冒泡

event.stopPropagation();

js 获取子节点

var getFirstChild = document.getElementById(“test”).children[0];
var getFirstChild = document.getElementById(“test”).firstElementChild;
var getLastChildB = document.getElementById(“test”).lastElementChild;

js 获取父节点

var p = document.getElementById(“test”).parentNode;

js获取兄弟节点

document.getElementById(“test”).previousElementSibling;
document.getElementById(“test”).nextElementSibling;

js获取屏幕宽高

body

document.body.clientWidth clientHeight 可见区域的宽高 内容高+padding
offsetWidth offsetHeight 内容高+padding+边框

window

window.innerHeight innerWidth 可见区域的宽高
window.screen.height width 分辨率
outerHeight outerWidth

scrollTop 滚动条卷曲长度
offsetTop 相对于版面或由 offsetTop 属性指定的父坐标的计算顶端位置

vim /etc/profile
export N_PREFIX=/opt/node-v8.0.0-linux-x64 #node实际安装位置
export PATH=$N_PREFIX/bin:$PATH
source /etc/profile
